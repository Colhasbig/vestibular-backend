function hash(texto) {
  return crypto.createHash('md5').update(texto).digest('hex');
}

async function safeJSON(callFn, retries = 2) {
  let lastError;
  for (let i = 0; i <= retries; i++) {
    try {
      const r = await callFn();
      return JSON.parse(r.choices[0].message.content);
    } catch (e) {
      lastError = e;
    }
  }
  throw lastError;
}
